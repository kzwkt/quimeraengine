# Convenciones de Documentación #

  * **Etiqueta summary**
```
/// <summary>
/// 
/// </summary>
```
  * **Etiqueta param**
```
/// <param name=""> [Opcional][IN]-[OUT]-[IN/OUT] </param>
```
  * **Etiqueta returns**
```
/// <returns>
/// 
/// </returns>
```
  * **Etiqueta exception**
```
/// <exception cref=""></exception>
```
  * **Enumeradores**
```
/// <summary>
///
/// </summary>
enum Enumerado
{
   Valor1, /*!< Descripción */
   Valor2  /*!< Descripción */
};
```
